<script>

const askBirthdayBtn = document.getElementById("askBirthdayBtn");

const birthdayMessage = document.getElementById("birthday-message");

const flowerCanvas = document.getElementById("flowerCanvas");

const ctx = flowerCanvas.getContext("2d");


askBirthdayBtn.addEventListener("click", () => {

const isBirthday = confirm("Es tu cumpleaños?");

if (isBirthday) {

drawFlower();

birthdayMessage.innerText = "Me tarde en hacer este detalle, pero ahora puedes decir que un ingeniero te desarrolló un código solo para ti.\nFeliz cumpleaños de chica guapa!";

} else {

birthdayMessage.innerText = "Oh, está bien! Espero que tengas un gran día de todas formas!";

}

});


function drawFlower() {

// Limpiar el canvas

ctx.clearRect(0, 0, flowerCanvas.width, flowerCanvas.height);


// Dibujar pétalos

ctx.strokeStyle = "#ff69b4";

ctx.lineWidth = 3;


// Pétalo superior

ctx.beginPath();

ctx.ellipse(200, 100, 50, 70, 0, 0, Math.PI * 2);

ctx.stroke();


// Pétalo izquierdo

ctx.beginPath();

ctx.ellipse(150, 200, 50, 70, Math.PI / 2, 0, Math.PI * 2);

ctx.stroke();


// Pétalo derecho

ctx.beginPath();

ctx.ellipse(250, 200, 50, 70, Math.PI / 2, 0, Math.PI * 2);

ctx.stroke();


// Pétalo inferior

ctx.beginPath();

ctx.ellipse(200, 300, 50, 70, 0, 0, Math.PI * 2);

ctx.stroke();


// Centro de la flor

ctx.fillStyle = "#FFD700";

ctx.beginPath();

ctx.arc(200, 200, 40, 0, Math.PI * 2);

ctx.fill();

}

</script>
